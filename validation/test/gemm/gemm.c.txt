/* gemm_int.c: integer GEMM variant for PolyBench/C */

#include <stdio.h>
#include <unistd.h>
#include <string.h>
#include <math.h>

/* Include polybench common header. */
#include "polybench.h"

/* Include benchmark-specific header (for NI/NJ/NK). */
#include "gemm.h"

/* Array initialization using int values. */
static
void init_array(int ni, int nj, int nk,
                int POLYBENCH_2D(C, NI, NJ, ni, nj),
                int POLYBENCH_2D(A, NI, NK, ni, nk),
                int POLYBENCH_2D(B, NK, NJ, nk, nj))
{
  int i, j;

  for (i = 0; i < ni; i++)
    for (j = 0; j < nj; j++)
      C[i][j] = (i * j) % 13;

  for (i = 0; i < ni; i++)
    for (j = 0; j < nk; j++)
      A[i][j] = (i * (j + 1)) % 17;

  for (i = 0; i < nk; i++)
    for (j = 0; j < nj; j++)
      B[i][j] = (i * (j + 2)) % 19;
}

/* DCE code. Must scan the entire live-out data. */
static
void print_array(int ni, int nj,
                 int POLYBENCH_2D(C, NI, NJ, ni, nj))
{
  int i, j;

  POLYBENCH_DUMP_START;
  POLYBENCH_DUMP_BEGIN("C");
  for (i = 0; i < ni; i++)
    for (j = 0; j < nj; j++) {
      if ((i * ni + j) % 20 == 0) fprintf(POLYBENCH_DUMP_TARGET, "\n");
      fprintf(POLYBENCH_DUMP_TARGET, "%d ", C[i][j]);
    }
  POLYBENCH_DUMP_END("C");
  POLYBENCH_DUMP_FINISH;
}

/* Main computational kernel. */
void kernel_int(int ni, int nj, int nk,
                int POLYBENCH_2D(C, NI, NJ, ni, nj),
                int POLYBENCH_2D(A, NI, NK, ni, nk),
                int POLYBENCH_2D(B, NK, NJ, nk, nj))
{
  int i, j, k;

  for (i = 0; i < _PB_NI; i++) {
    for (k = 0; k < _PB_NK; k++) {
      // #pragma clang loop vectorize(disable) unroll_count(4)
      for (j = 0; j < _PB_NJ; j++)
        C[i][j] += A[i][k] * B[k][j];
    }
  }
}

int main(int argc, char** argv)
{
  int ni = NI;
  int nj = NJ;
  int nk = NK;

  POLYBENCH_2D_ARRAY_DECL(C, int, NI, NJ, ni, nj);
  POLYBENCH_2D_ARRAY_DECL(A, int, NI, NK, ni, nk);
  POLYBENCH_2D_ARRAY_DECL(B, int, NK, NJ, nk, nj);

  init_array(ni, nj, nk,
             POLYBENCH_ARRAY(C),
             POLYBENCH_ARRAY(A),
             POLYBENCH_ARRAY(B));

  polybench_start_instruments;

  kernel_int(ni, nj, nk,
             POLYBENCH_ARRAY(C),
             POLYBENCH_ARRAY(A),
             POLYBENCH_ARRAY(B));

  polybench_stop_instruments;
  polybench_print_instruments;

  polybench_prevent_dce(print_array(ni, nj, POLYBENCH_ARRAY(C)));

  POLYBENCH_FREE_ARRAY(C);
  POLYBENCH_FREE_ARRAY(A);
  POLYBENCH_FREE_ARRAY(B);

  return 0;
}
